需要用到的包为 

    npm i -D electron-packager
在package.json的script 中进行设置,可做参考 

    "package:win32": "electron-packager ./ ele_demo --app-version=1.0.4  --platform=win32  --arch=ia32 --overwrite --out=out --icon=./assets/img/ele.ico",
    "package:win64": "electron-packager ./ ele_demo --app-version=1.0.4  --platform=win32  --arch=x64  --overwrite --out=out --icon=./assets/img/ele.ico",
    "package:linux": "electron-packager ./ ele_demo --app-version=1.0.4  --platform=linux  --arch=x64  --overwrite --out=out --icon=./assets/img/ele.ico",
    "package:mac":   "electron-packager ./ ele_demo --app-version=1.0.4  --platform=darwin --arch=x64  --overwrite --out=out --icon=./assets/img/ele.ico",
    "package:all":   "electron-packager ./ ele_demo --app-version=1.0.4  --all                         --overwrite --out=out --icon=./assets/img/ele.ico"
打包成加密 
npm i -D asar
配置 package.json 
    "asar:win32": "asar pack out/package-win32-ia32/resources/app/ out/package-win32-ia32/resources/app.asar"
生成安装器
